########## 아이템을 가지고 있는 경우 ##########

block Item 18228 >= 1 || Item 18229 >= 1 || ItemSpace == 0

	#이 앞에는 우귀가 기르는 개와 우귀가 있어··. 어떻게 도망칠까.
	EndWindow 11225


########## 아이템을 가지고 있지 않은 경우 ##########

block Item 18228 == 0 && Item 18229 == 0

	#주, 죽는 줄 알았어! 후우, 살려줘서 고맙소. 이 편지를 아버지 타일러에게 전해주시오. 보답을 해줄 겁니다.
	Window 11226,OK

	Window_OK ItemSpace >= 1
	GiveItem 18229

	Window_OK ItemSpace == 0
	EndWindow 10000
